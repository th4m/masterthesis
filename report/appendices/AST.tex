\chapter{Abstract Syntax Tree of Expressions}
\label{app:ASTExp}

\begin{figure}
\begin{alltt}
Exp   = Raise Exp
      | Handle Exp [(Pat, Exp)]
      | Literal Lit
      | Con (Maybe (Id ConN)) [Exp]
      | Var (Id VarN)
      | Fun VarN Exp
      | App Op [Exp]
      | Log LOp Exp Exp
      | If Exp Exp Exp
      | Mat Exp [(Pat, Exp)]
      | Let (Maybe VarN) Exp Exp
      | LetRec [(VarN, VarN, Exp)] Exp
      | TAnnot Exp T
\end{alltt}
\caption{Expressions of CakeML}
\end{figure}

\begin{figure}
\begin{alltt}
data Op
  -- Integer operations
  = OPN Opn
  | OPB Opb
  -- Equality
  | Equality
  -- Function application
  | OpApp
  -- Char operations
  | Ord
  | Chr
  | ChOpb Opb
  -- String operations
  | Implode
  | StrSub
  | StrLen
  -- Vector operations
  | VFromList
  | VSub
  | VLength
\end{alltt}
\caption{Operations for the App constructor of expressions}
\end{figure}

\begin{figure}
\begin{alltt}
data Opn

  | Times
  | Divide
  | Modulo
\end{alltt}
\caption{Numerical operators for the App constructor of expressions}
\end{figure}
